# Internationalization (i18n)

## Stack

- **@lingui/react** + **@lingui/core** — Runtime translation
- **@lingui/macro** — Compile-time message extraction
- **@lingui/cli** — CLI for extracting and compiling messages
- **@lingui/babel-plugin-lingui-macro** — Babel plugin for macro transformation
- **expo-localization** — Device locale detection

## Lingui Configuration

```typescript
// lingui.config.ts

import type { LinguiConfig } from '@lingui/conf';

const config: LinguiConfig = {
  locales: ['en', 'uk'],  // Add supported locales
  sourceLocale: 'en',
  catalogs: [
    {
      path: 'src/locale/locales/{locale}/messages',
      include: ['src'],
    },
  ],
  format: 'ts',
};

export default config;
```

## i18n Setup

```typescript
// src/locale/constants.ts

export type AppLocale = 'en' | 'uk';

export const SUPPORTED_LOCALES: Record<AppLocale, string> = {
  en: 'English',
  uk: 'Українська',
};

export const DEFAULT_LOCALE: AppLocale = 'en';
```

```typescript
// src/locale/i18n.ts

import { i18n } from '@lingui/core';
import { getLocales } from 'expo-localization';
import { clientStorage } from '@/store/storage';
import { STORAGE_KEYS } from '@/constants/keys';
import { DEFAULT_LOCALE, SUPPORTED_LOCALES } from './constants';
import type { AppLocale } from './constants';

export const getDeviceLocale = (): AppLocale => {
  const deviceLocales = getLocales();
  const deviceLang = deviceLocales[0]?.languageCode as AppLocale;

  if (deviceLang && deviceLang in SUPPORTED_LOCALES) {
    return deviceLang;
  }

  return DEFAULT_LOCALE;
};

export const getStoredLocale = (): AppLocale | null =>
  clientStorage.getString(STORAGE_KEYS.LANGUAGE) as AppLocale | null;

export const setStoredLocale = (locale: AppLocale): void =>
  clientStorage.setString(STORAGE_KEYS.LANGUAGE, locale);

export const dynamicActivate = async (locale: AppLocale) => {
  const { messages } = await import(`./locales/${locale}/messages.ts`);
  i18n.load(locale, messages);
  i18n.activate(locale);
};

export const initI18n = async () => {
  const stored = getStoredLocale();
  const locale = stored ?? getDeviceLocale();
  await dynamicActivate(locale);
  return locale;
};

export { i18n };
```

## i18n Provider

```typescript
// src/locale/i18n-provider.tsx

import { useEffect, useState } from 'react';
import { I18nProvider as LinguiProvider } from '@lingui/react';
import { i18n, initI18n } from './i18n';
import type { AppLocale } from './constants';

export const I18nProvider = ({ children }: { children: React.ReactNode }) => {
  const [isReady, setIsReady] = useState(false);

  useEffect(() => {
    initI18n().then(() => setIsReady(true));
  }, []);

  if (!isReady) return null;

  return <LinguiProvider i18n={i18n}>{children}</LinguiProvider>;
};
```

## Usage in Components

```typescript
import { useLingui } from '@lingui/react/macro';
import { Trans } from '@lingui/react/macro';

const MyScreen = () => {
  const { t } = useLingui();

  return (
    <View>
      {/* Simple string */}
      <Text>{t`Welcome back`}</Text>

      {/* With JSX */}
      <Trans>
        <Text>Hello, <Text style={styles.bold}>{userName}</Text></Text>
      </Trans>

      {/* Plurals */}
      <Text>{t`${count} item${count === 1 ? '' : 's'}`}</Text>

      {/* As prop value */}
      <Button title={t`Save`} />
      <TextInput placeholder={t`Enter your name`} />

      {/* Accessibility — always i18n */}
      <Pressable
        accessibilityLabel={t`Play`}
        accessibilityHint={t`Plays the selected track`}
        accessibilityRole="button"
      />
    </View>
  );
};
```

## Language Switching

```typescript
// In a Language Selection screen:
import { dynamicActivate, setStoredLocale } from '@/locale/i18n';
import { SUPPORTED_LOCALES, type AppLocale } from '@/locale/constants';

const LanguageScreen = () => {
  const handleLanguageChange = async (locale: AppLocale) => {
    await dynamicActivate(locale);
    setStoredLocale(locale);
  };

  return (
    <FlatList
      data={Object.entries(SUPPORTED_LOCALES)}
      renderItem={({ item: [locale, label] }) => (
        <Pressable onPress={() => handleLanguageChange(locale as AppLocale)}>
          <Text>{label}</Text>
        </Pressable>
      )}
    />
  );
};
```

## CLI Commands

Add to `package.json` scripts:

```json
{
  "scripts": {
    "intl:extract": "lingui extract --locale en",
    "intl:extract:all": "lingui extract",
    "intl:compile": "lingui compile --typescript"
  }
}
```

**Workflow:**
1. Write components using `t` tagged template or `<Trans>` components
2. Run `npm run intl:extract:all` — extracts all translatable strings
3. Translate strings in `src/locale/locales/[lang]/messages.ts`
4. Run `npm run intl:compile` — compiles for production
5. Commit the compiled message files

## Rules

1. **Never hardcode user-facing strings** — always use `t` or `<Trans>`, including `accessibilityLabel`, `accessibilityHint`, `placeholder`, and `title` props
2. **Extract after adding new strings** — run `intl:extract:all`
3. **Store language preference in MMKV** — not AsyncStorage
4. **Detect device locale on first launch** — then respect user's manual choice
5. **Lazy load locale data** — use `dynamicActivate` to load only needed locale
